        }

    }

    public class WorkConsumer {

    public static void main(String[] args) throws IOException, TimeoutException {

        //创建连接
        Connection connection = ConnectionUtil.getConnection();

        try{
            //创建通道

            //队列声明，不声明也可以
            channel.queueDeclare("work_queue", false, false,false, null);

            DefaultConsumer consumer = new DefaultConsumer(channel){
                @Override
                public void handleDelivery(String consumerTag, Envelope envelope, AMQP.BasicProperties properties, byte[] body) throws IOException {
                    System.out.println("recv[1]: "+new String(body, StandardCharsets.UTF_8));
                }
            };

            boolean autoAck = false;//
            //监听队列
            channel.basicConsume("work_queue", autoAck, consumer);
            TimeUnit.SECONDS.sleep(1);

        }catch (Exception e){
            e.printStackTrace();
        }
    }

    public class WorkConsumer2 {

    public static void main(String[] args) throws IOException, TimeoutException {

        //创建连接
        Connection connection = ConnectionUtil.getConnection();

        try{
            //创建通道
            Channel channel = connection.createChannel();

            //队列声明，不声明也可以
            channel.queueDeclare("work_queue", false, false,false, null);

            DefaultConsumer consumer = new DefaultConsumer(channel){
                @Override
                public void handleDelivery(String consumerTag, Envelope envelope, AMQP.BasicProperties properties, byte[] body) throws IOException {
                    System.out.println("recv[2]: " + new String(body, StandardCharsets.UTF_8));
                    try {
                        TimeUnit.SECONDS.sleep(5);
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }
            };
		boolean autoAck = false;
            //监听队列  autoAck 为true  轮询分发（公平分发）  反之为限制分发
            channel.basicConsume("work_queue", autoAck, consumer);
        }catch (Exception e){
            e.printStackTrace();
        }
    }

    # 轮询分发

    1. 消息队列并不清楚各队列处理数据的能力（速度快慢）
    2. 无论队列处理快慢，进行公平分发，所有队列处理的数据量相同

    # 限制分发

    1. 生产者、消费者设置 channel.basicQos(1); 表示消费者回复确认之前，生产者每次最多发一条数据给消费者

      存疑，生产者是否必须设置？

    1. 消费者需关闭自动确认
    2. 处理能力高的队列处理更多的数据
